data() {
  return {
    formConfig: {
      httpConfig: (config)=>{
        // 这里自定会议header的请求参数
        config.headers['aaaa'] = 'bbbb'
        return config
      }
    }

  }

}

 <VueDragFormBuild ref="formbuild"  :config="formConfig"/>
